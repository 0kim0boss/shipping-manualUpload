    function handleFileUpload(event) {

      var file = event.target.files[0];

      if (!file) return;


      var reader = new FileReader();

      reader.onload = function (e) {

        try {

          var data = new Uint8Array(e.target.result);

          var workbook = XLSX.read(data, { type: 'array' });

          var sheet = workbook.Sheets[workbook.SheetNames[0]];

          var rows = XLSX.utils.sheet_to_json(sheet, { header: 1, defval: "" });


          if (rows.length < 2) {

            alert("데이터가 없습니다. 헤더 포함 최소 2행이 필요합니다.");

            return;

          }


          // 사용자 선택 배송사

          var selectedCarrier = document.getElementById('manualCarrier').value;

          currentCarrier = selectedCarrier;


          // 국가 코드 변환 맵

          var cMap = sheetData.countryMap || {};

          cMap["north macedonia"] = "MK";

          cMap["republic of north macedonia"] = "MK";

          var sMap = sheetData.stateMap || {};

          var bMap = sheetData.boxMap || {};


          // 컬럼 인덱스 (0-based, 사용자 지정 매핑)

          // B(1):Order no, G(6):Product, H(7):Price, I(8):Qty, J(9):Ship via

          // K(10):Name, L(11):Phone, M(12):Country, N(13):Province, O(14):City

          // P(15):Addr1, Q(16):Addr2, R(17):Zip, S(18):Email, T(19):TaxID

          // U(20):Weight, V(21):Box Type

          var COL = {

            orderNo: 0, product: 6, price: 7, qty: 8, shipVia: 9,

            name: 10, phone: 11, country: 12, province: 13, city: 14,

            addr1: 15, addr2: 16, zip: 17, email: 18, taxId: 19,

            weight: 20, boxType: 21

          };


          // 주문번호 기준 그룹핑

          var grouped = {};

          for (var i = 1; i < rows.length; i++) {

            var row = rows[i];

            var orderNo = String(row[COL.orderNo] || "").trim();

            if (!orderNo) continue;


            // 국가 코드 변환

            var rawCountry = String(row[COL.country] || "").trim();

            var countryKey = rawCountry.toLowerCase().replace(/\s+/g, " ").trim();

            var isoCountry = "";

            var countryError = "";

            if (/^[A-Za-z]{2}$/.test(rawCountry)) {

              isoCountry = rawCountry.toUpperCase();

            } else if (cMap[countryKey]) {

              isoCountry = String(cMap[countryKey]).trim().toUpperCase();

            } else {

              countryError = "Country mapping failed: " + rawCountry;

            }


            if (!grouped[orderNo]) {

              // J열(Ship via)에서 배송사 파싱 - K-packet, 우체국 등

              var shipViaRaw = String(row[COL.shipVia] || "").trim().toUpperCase();

              var parsedCarrier = selectedCarrier;

              if (shipViaRaw.includes("K-PACKET") || shipViaRaw.includes("우체국") || (shipViaRaw.includes("KOREA") && shipViaRaw.includes("POST"))) {

                parsedCarrier = "KOREA_POST";

              } else if (shipViaRaw.includes("ACI")) {

                parsedCarrier = "ACI";

              } else if (shipViaRaw.includes("FEDEX") || shipViaRaw.includes("DHL") || shipViaRaw.includes("UPS") || shipViaRaw.includes("EMS") || shipViaRaw.includes("ETC")) {

                parsedCarrier = "EXPRESS_PENDING";

              }


              // 박스 타입 → 치수 자동 매핑

              var boxCode = String(row[COL.boxType] || "").trim().toUpperCase();

              var boxDim = bMap[boxCode] || { length: 20, width: 20, height: 10 };


              // 무게 (g→kg 변환: 값이 10 이상이면 g으로 간주)

              var rawWeight = parseFloat(row[COL.weight]) || 0.5;

              var weightKg = (rawWeight >= 10) ? rawWeight / 1000 : rawWeight;

              if (weightKg <= 0) weightKg = 0.5;


              // State(Province) 코드 변환

              var rawState = String(row[COL.province] || "").trim();

              var mappedState = rawState;

              if (rawState.length === 2) {

                mappedState = rawState.toUpperCase();

              } else if (["US", "CA", "IN", "MX"].indexOf(isoCountry) > -1 && sMap[rawState.toLowerCase().replace(/\s+/g, " ").trim()]) {

                mappedState = sMap[rawState.toLowerCase().replace(/\s+/g, " ").trim()];

              }


              grouped[orderNo] = {

                orderNo: orderNo,

                carrier: parsedCarrier,

                trackingNo: "",

                error: countryError,

                eventName: "",

                receiver: {

                  name: String(row[COL.name] || ""),

                  phone: String(row[COL.phone] || ""),

                  email: String(row[COL.email] || ""),

                  country: isoCountry,

                  state: mappedState,

                  city: String(row[COL.city] || ""),

                  address1: String(row[COL.addr1] || ""),

                  address2: String(row[COL.addr2] || ""),

                  postalCode: String(row[COL.zip] || ""),

                  taxId: String(row[COL.taxId] || ""),

                  incoterms: "DAP"

                },

                package: {

                  weight: weightKg,

                  boxCode: boxCode,

                  length: parseFloat(boxDim.length) || 20,

                  width: parseFloat(boxDim.width) || 20,

                  height: parseFloat(boxDim.height) || 10

                },

                shipper: getBlankManualShipper(),

                items: []

              };

            }


            // 상품 추가

            var itemDesc = String(row[COL.product] || "Cosmetics").trim();

            var itemPrice = parseFloat(row[COL.price]) || 1;

            var itemQty = parseInt(row[COL.qty]) || 1;


            grouped[orderNo].items.push({

              description: itemDesc,

              price: itemPrice,

              quantity: itemQty,

              sku: "",

              hsCode: "330499",

              originCountry: "KR"

            });

          }


          // EXPRESS_PENDING이 있으면 운송사 선택 모달 표시

          var hasExpressPending = false;

          for (var k in grouped) { if (grouped[k].carrier === "EXPRESS_PENDING") { hasExpressPending = true; break; } }


          if (hasExpressPending) {

            expressCarrierPendingGrouped = grouped;

            document.getElementById('expressCarrierModal').classList.remove('hidden');

            return;

          }


          finishManualUpload(grouped);


        } catch (err) {

          console.error("파일 파싱 오류:", err);

          alert("파일 파싱 중 오류가 발생했습니다: " + err.message);

        }

      };


      reader.readAsArrayBuffer(file);

      // 같은 파일 재업로드 가능하도록 초기화

      event.target.value = "";

    }

    function renderItemTable(items) {

      var con = document.getElementById('itemContainer');

      var html = '<table class="item-table"><thead><tr><th>상품명 (Description)</th><th width="80">수량</th><th width="100">단가($)</th><th width="100" class="text-right">총액</th><th width="40"></th></tr></thead><tbody>';

      var total = 0;


      items.forEach(function (i, idx) {

        var price = parseFloat(i.price) || 0;

        var qty = parseInt(i.quantity) || 0;

        var t = price * qty;

        total += t;

        var descEsc = (i.description || "").replace(/"/g, "&quot;");

        html += '<tr data-item-idx="' + idx + '">';

        html += '<td><input value="' + descEsc + '" class="i-desc"></td>';

        html += '<td><input value="' + qty + '" class="i-qty" type="number"></td>';

        html += '<td><input value="' + price + '" class="i-price" type="number"></td>';

        html += '<td class="px-3 py-2 text-right text-slate-600 font-medium">$' + t.toFixed(2) + '</td>';

        html += '<td class="px-1"><button type="button" class="p-1 text-slate-400 hover:text-red-600 hover:bg-red-50 rounded" onclick="removeItemRow(this)" title="상품 제거">&#10005;</button></td>';

      con.innerHTML = html;

    }

      async function runConcurrent(tasks, limit) {

        var results = [];

        var executing = [];

        for (const task of tasks) {

          const p = task().then(res => {

            results.push(res);

            executing.splice(executing.indexOf(p), 1);

            return res; // Return result for chaining

          });

          executing.push(p);

          if (executing.length >= limit) {

            await Promise.race(executing);

          }

        }

        await Promise.all(executing);

        return results;

      }


      // 3. 각 청크를 처리하는 함수 생성

      var tasks = chunks.map(chunkIds => async () => {

        var allOrders = getActiveOrders();

        var chunkData = chunkIds.map(idx => {

          var o = JSON.parse(JSON.stringify(allOrders[idx]));

          o.shipper = o.shipper || sheetData.shipper;

          if (forceRepublish) o.forceRepublish = true;

          return o;

            google.script.run

              .withSuccessHandler(resolve)

              .withFailureHandler(reject)

          });

                var targetBatch = bulkDocsMap[currentCarrier].batches.find(b => b.id === batchId);

                if (targetBatch) targetBatch.docs = targetBatch.docs.concat(labels);


                var originalIdx = activeOrders.findIndex(o => o.orderNo == r.orderNo);

                if (originalIdx > -1) {

                  activeOrders[originalIdx].trackingNo = r.trackingNo;

                }

                generatedResults.push(r.trackingNo);

              } else {

                var originalIdx = activeOrders.findIndex(o => o.orderNo == r.orderNo);

                if (originalIdx > -1) {

                  activeOrders[originalIdx].error = r.message;

                }

              }

            });

          }

        } catch (e) { console.error("Bulk Error:", e); }


        // UI 업데이트 (순차 애니메이션 보장)

        // uiUpdateQueue를 사용하여 UI 업데이트가 순차적으로 실행되도록 보장

        uiUpdateQueue = uiUpdateQueue.then(async () => {

          var startCount = uiProcessed;

          var endCount = startCount + chunkIds.length;


          for (var c = startCount + 1; c <= endCount; c++) {

            document.getElementById('loadingText').innerText = `발행 중... (${c} / ${total})`;

            btn.innerText = `(${c}/${total})`;

            await new Promise(r => setTimeout(r, 50)); // 속도감을 위해 0.05초로 단축

          renderOrderList(); // 렌더링 갱신

        });

        await uiUpdateQueue; // 현재 UI 업데이트가 완료될 때까지 대기

      });


      // 4. 실행

      await runConcurrent(tasks, CONCURRENCY_LIMIT);


      // 완료 상태를 잠시 보여주기 위한 대기 (0.5초)

      await new Promise(r => setTimeout(r, 500));


      window.onbeforeunload = null;

    async function downloadMergedPdf(batchId) {

      var data = bulkDocsMap[currentCarrier];

      if (!data || !data.batches) return;


      // 특정 배치만 선택 or 전체(지금은 배치별 다운로드만 구현 요청됨)

      var targetDocs = [];

      if (batchId) {

        var batch = data.batches.find(b => b.id === batchId);

        if (batch) targetDocs = batch.docs;

      }


      if (targetDocs.length === 0) { alert("다운로드할 문서가 없습니다."); return; }


      var btn = document.querySelector('.btn-download'); // 필요시 로딩 표시용


      const { PDFDocument, degrees } = PDFLib;

      const mergedPdf = await PDFDocument.create();


      for (const doc of targetDocs) {

        if (currentCarrier === 'UPS') {

          const pngUrl = await convertGifToPng(doc.base64);

          const pngImage = await mergedPdf.embedPng(pngUrl);

          const { width, height } = pngImage;


          const page = mergedPdf.addPage([288, 432]);


          if (width > height) {

            page.drawImage(pngImage, {

              x: 288, y: 0, width: 432, height: 288, rotate: degrees(90),

            });

          } else {

            page.drawImage(pngImage, { x: 0, y: 0, width: 288, height: 432 });

          }

        } else {

          var b64 = doc.base64.replace(/^data:application\/pdf;base64,/, "");

          const pdf = await PDFDocument.load(b64);

          const pages = await mergedPdf.copyPages(pdf, pdf.getPageIndices());

          pages.forEach(p => mergedPdf.addPage(p));

        }

      }


      const pdfBytes = await mergedPdf.save();

      const blob = new Blob([pdfBytes], { type: 'application/pdf' });

      const link = document.createElement('a');

      link.href = URL.createObjectURL(blob);

      link.download = `Merged_${currentCarrier}_Labels.pdf`;

      document.body.appendChild(link);

      link.click();

      document.body.removeChild(link);


      if (btn) btn.innerText = orgText;

    }


    function convertGifToPng(base64Gif) {

      return new Promise((resolve) => {

        var img = new Image();

        img.onload = function () {

          var canvas = document.createElement('canvas');

          canvas.width = img.width;

          canvas.height = img.height;

          var ctx = canvas.getContext('2d');

          ctx.drawImage(img, 0, 0);

          resolve(canvas.toDataURL('image/png'));

        };

        img.src = "data:image/gif;base64," + base64Gif.replace(/^data:image\/gif;base64,/, "");

      });

    }

    window.renderProductSummary = function (data) {

      var tbody = document.getElementById('popup-summary-body');

      var countEl = document.getElementById('popup-total-count');

      var sumEl = document.getElementById('popup-total-sum');


      if (!data || data.length === 0) {

        tbody.innerHTML = '<tr><td colspan="2" class="p-8 text-center text-slate-400 text-xs">데이터가 없습니다.</td></tr>';

        if (countEl) countEl.innerText = "0";

        if (sumEl) sumEl.innerText = "0";

        return;

      }


      var html = "";

      var totalQty = 0;


      data.forEach(item => {

        html += `<tr class="border-b border-slate-50 hover:bg-slate-50 transition-colors">

             <td class="px-4 py-2 text-slate-600 leading-tight break-words py-3" title="${item.product}">${item.product}</td>

             <td class="px-4 py-2 text-right font-medium text-slate-700 align-top py-3">${item.totalQty}</td>

           </tr>`;

        totalQty += item.totalQty;

      });


      tbody.innerHTML = html;

      if (countEl) countEl.innerText = data.length;

      if (sumEl) sumEl.innerText = totalQty.toLocaleString();

    };

    function downloadCsv() {

      showLoading("CSV 생성 중...");

      google.script.run

        .withSuccessHandler(function (data) {

          hideLoading();

          if (!data || data.length === 0) { alert("다운로드할 데이터가 없습니다."); return; }


          var csv = "\uFEFFOrder No,Response\n";

          data.forEach(function (r) {

            var cleanOrder = String(r.orderNo).replace(/,/g, " ").trim();

            var cleanRes = String(r.response).replace(/,/g, " ").replace(/\n/g, " ").trim();

            csv += cleanOrder + "," + cleanRes + "\n";

          });


          var blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });

          var link = document.createElement("a");

          link.href = URL.createObjectURL(blob);

          link.download = "Order_Response_List.csv";

          document.body.appendChild(link);

          link.click();

          document.body.removeChild(link);

        })

        .withFailureHandler(function (error) {

          hideLoading();

          alert("CSV 다운로드 실패: " + error);

        })

        .getDataForCsv(userSlot); // [수정] 슬롯 전달

    }
